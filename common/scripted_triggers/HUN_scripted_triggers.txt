#scripted trigger for Hungary

is_available_fighter_HUN = {
	OR = {
		AND = {
			has_tech = fighter1
			TST = { 
				NOT = { has_tech = fighter1}
			}
		}
		AND = {
			has_tech = fighter2
			TST = { 
				NOT = { has_tech = fighter2}
			}
		}
		AND = {
			has_tech = fighter3
			TST = { 
				NOT = { has_tech = fighter3}
			}
		}
	}
	NOT = {
		has_war_with = TST
	}
}
